# Helper function to calculate stats for a given dataset
calculate_stats <- function(data, fattyacid) {
  mean_val <- mean(data[[fattyacid]], na.rm = TRUE)
  sd_val <- sd(data[[fattyacid]], na.rm = TRUE)
  mean_sd <- paste0(round(mean_val, 1), " ± ", round(sd_val, 1))
  return(c(mean = round(mean_val, 3), sd = round(sd_val, 3), mean_sd = mean_sd))
}

# Helper function to format p-values
format_p_value <- function(p_value) {
  if (p_value < 0.001) {
    return("P<0.001")
  } else {
    return(round(p_value, 5))
  }
}

# Loop through each fatty acid
for (fattyacid in fattyacidsToTest) {
  # Create the formula with both site2, weight, and their interaction as predictors
  formula <- as.formula(paste0("`", fattyacid, "` ~ site2 * weight"))
  
  # Fit the linear model
  mod <- lm(formula, data = wholecont_transformed)
  
  # Extract p-values for site2, weight, and their interaction
  p_value_site2 <- format_p_value(anova(mod)["site2", "Pr(>F)"])
  p_value_weight <- format_p_value(anova(mod)["weight", "Pr(>F)"])
  p_value_interaction <- format_p_value(anova(mod)["site2:weight", "Pr(>F)"])
  
  # Calculate R-squared value
  r_squared <- round(r2(mod)$R2, 3)

  # Extract F-statistic
  f_stat <- summary(mod)$fstatistic
  f_stat_string <- paste0(round(f_stat["value"], 2), " on ", f_stat["numdf"], " and ", f_stat["dendf"], " DF")
  
  # Perform Tukey post-hoc test and calculate significance letters
  # Perform Tukey post-hoc test and calculate significance letters
  anova_model <- aov(as.formula(paste0("`", fattyacid, "` ~ site2")), data = wholecont_transformed)
  tmp <- -1
  tryCatch({
    tukey <- TukeyHSD(anova_model)
    cld_results <- glht(anova_model, linfct = mcp(site2 = "Tukey"))
    tmp <- cld(cld_results, level = 0.10)$mcletters$Letters
    tukey_wild <- tmp["wild"]
    tukey_sea_cage <- tmp["sea cage"]
    tukey_land_based <- tmp["land based"]
  }, error = function(e) {
    tukey_wild <- tukey_sea_cage <- tukey_land_based <- NA
  })
  
  # Append the letters to the list
  letters[[fattyacid]] <- tmp

  # Calculate stats for the full dataset
  all_stats <- calculate_stats(wholecont, fattyacid)
  
  # Calculate stats for each site2 category
  wild_stats <- calculate_stats(wholecont[wholecont$site2 == "wild", ], fattyacid)
  sea_cage_stats <- calculate_stats(wholecont[wholecont$site2 == "sea cage", ], fattyacid)
  land_based_stats <- calculate_stats(wholecont[wholecont$site2 == "land based", ], fattyacid)
  
  # Append the results to the list as a named vector
  results[[fattyacid]] <- c(
    fattyacid = fattyacid,
    all_stats,
    p_site2 = p_value_site2,
    p_weight = p_value_weight,
    p_interaction = p_value_interaction,
    r_squared = r_squared,
    f_statistic = f_stat_string,
    tukey_wild = tmp["wild"],

# Filter the dataset to only include sea cage fish
sea_cage_data <- wholecont[wholecont$site2 == "sea cage", ]

# Ensure season names are correctly capitalized and remain a factor
sea_cage_data$season <- factor(sea_cage_data$season, levels = c("Winter", "Summer", "Autumn"))

results <- list()
letters <- list()

# Helper function to calculate stats for a given dataset
calculate_stats <- function(data, fatty_acid) {
  if (all(is.na(data[[fatty_acid]]))) {
    return(c(mean=NA, sd=NA, mean_sd="NA"))
  }
  mean_val <- mean(data[[fatty_acid]], na.rm = TRUE)
  sd_val <- sd(data[[fatty_acid]], na.rm = TRUE)
  mean_sd <- paste0(round(mean_val, 2), " ± ", round(sd_val, 2))
  return(c(mean = round(mean_val, 3), sd = round(sd_val, 3), mean_sd = mean_sd))
}

# Helper function to format p-values
format_p_value <- function(p_value) {
  if (is.na(p_value)) {
    return("NA")
  } else if (p_value < 0.001) {
    return("P<0.001")
  } else {
    return(round(p_value, 5))
  }
}

# Loop through each fatty acid
for (fatty_acid in fattyacidsToTest) {
  # Check if there are at least two unique seasons with valid data
  valid_seasons <- unique(sea_cage_data$season[!is.na(sea_cage_data[[fatty_acid]])])
  
  if (length(valid_seasons) < 2) {
    # Skip fatty acids where season has only one level
    message(paste("Skipping", fatty_acid, "- insufficient season levels"))
    next
  }
  
  # Create the formula with season and weight as predictors
  formula <- as.formula(paste0("`", fatty_acid, "` ~ season * weight"))
  
  # Fit the linear model
  mod <- lm(formula, data = sea_cage_data)
  
  # Extract p-values for season, weight, and their interaction
  p_value_season <- format_p_value(anova(mod)["season", "Pr(>F)"])
  p_value_weight <- format_p_value(anova(mod)["weight", "Pr(>F)"])
  p_value_interaction <- format_p_value(anova(mod)["season:weight", "Pr(>F)"])
  
  # Calculate R-squared value
  r_squared <- round(r2(mod)$R2, 3)

  # Extract F-statistic
  f_stat <- summary(mod)$fstatistic
  f_stat_string <- paste0(round(f_stat["value"], 2), " on ", f_stat["numdf"], " and ", f_stat["dendf"], " DF")
  
  # Perform Tukey HSD post-hoc test and calculate significance letters
  tukey_letters <- c(Winter = NA, Summer = NA, Autumn = NA)  # Default NA in case Tukey fails
  
  tryCatch({
    anova_model <- aov(as.formula(paste0("`", fatty_acid, "` ~ season")), data = sea_cage_data)
    tukey <- TukeyHSD(anova_model)
    cld_results <- glht(anova_model, linfct = mcp(season = "Tukey"))
    tukey_letters <- cld(cld_results, level = 0.10)$mcletters$Letters  # Extract Tukey grouping
  }, error = function(e) {
  })
  
  # Store Tukey letters
  letters[[fatty_acid]] <- tukey_letters
  
  # Calculate stats for the full dataset
  all_stats <- calculate_stats(sea_cage_data, fatty_acid)
  
  # Calculate stats for each available season
  winter_stats <- if ("Winter" %in% valid_seasons) calculate_stats(sea_cage_data[sea_cage_data$season == "Winter", ], fatty_acid) else c(mean=NA, sd=NA, mean_sd="NA")
  summer_stats <- if ("Summer" %in% valid_seasons) calculate_stats(sea_cage_data[sea_cage_data$season == "Summer", ], fatty_acid) else c(mean=NA, sd=NA, mean_sd="NA")
  autumn_stats <- if ("Autumn" %in% valid_seasons) calculate_stats(sea_cage_data[sea_cage_data$season == "Autumn", ], fatty_acid) else c(mean=NA, sd=NA, mean_sd="NA")
  
  # Append the results to the list as a named vector
  results[[fatty_acid]] <- c(
    fatty_acid = fatty_acid,
    all_stats,
    p_season = p_value_season,
    p_weight = p_value_weight,
    p_interaction = p_value_interaction,
    r_squared = r_squared,
    f_statistic = f_stat_string,
    tukey_winter = tukey_letters["Winter"],
    tukey_summer = tukey_letters["Summer"],
    tukey_autumn = tukey_letters["Autumn"],
    winter_mean = winter_stats["mean"],
    winter_sd = winter_stats["sd"],
    winter_mean_sd = winter_stats["mean_sd"],
    summer_mean = summer_stats["mean"],
    summer_sd = summer_stats["sd"],
    summer_mean_sd = summer_stats["mean_sd"],
    autumn_mean = autumn_stats["mean"],
    autumn_sd = autumn_stats["sd"],
    autumn_mean_sd = autumn_stats["mean_sd"]
  )
}
